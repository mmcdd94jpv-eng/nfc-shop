function generateReview(keyword){

  const resultBox =
  document.getElementById("reviewResult");

  const data = reviews[keyword];

  const start =
  data.start[Math.floor(Math.random()*data.start.length)];

  const middle1 =
  data.middle[Math.floor(Math.random()*data.middle.length)];

  const middle2 =
  data.middle[Math.floor(Math.random()*data.middle.length)];

  const end =
  data.end[Math.floor(Math.random()*data.end.length)];

  const finalText =
  start + middle1 + middle2 + end;

  resultBox.innerHTML = finalText;

  navigator.clipboard.writeText(finalText);

  alert("随机评价已自动复制");

}

function copyReview(){

  const text =
  document.getElementById("reviewResult").innerText;

  navigator.clipboard.writeText(text);

  alert("评价已复制");

}
